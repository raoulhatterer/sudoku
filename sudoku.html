2 Sacs de chiffres restant à placer.

 1: class sac:
 2:     """
 3:     Classe représentant un sac. Un sac contient des symboles identiques.
 4: 
 5:     Un sac contient à l'origine 9 symboles identiques.
 6:     À l'initialisation de la grille, un certain nombre de ces symboles est
 7:     prélevé pour les placer sur la grille.
 8:     Au cours du jeu, tant qu'il reste des symboles dans un sac, le joueur peut
 9:     en piocher pour les placer sur la grille.
10: 
11:     Attributs:
12:     ---------
13:     symbole : symbole placé dans le sacs
14:     cardinal : nombre de symboles que le sac contient
15:     """
16:     def __init__(self, symbole):
17:         """
18:         Constructeur de sac contenant 9 symboles identiques
19:         """
20:         self.symbole = symbole
21:         self.cardinal = 9  # nombre d'éléments dans le sac
22: 
23:     def __repr__(self):
24:         """
25:         Représentation d'un sac lorsque l'on tape son nom dans l'interpréteur.
26: 
27:         Le contenu du sac est renvoyé.
28:         """
29:         if self.cardinal == 0:
30:             return "sac vide"
31:         elif self.cardinal == 1:
32:             return "contient 1 chiffre {}".format(self.symbole)
33:         else:
34:             return "contient {} chiffres {}".format(self.cardinal, self.symbole)
35: 
36: 
37: class pioche():
38:     """
39:     Classe représentant 9 sacs contenant chacun des symboles identiques
40:     tous différents (1 sac avec que des "1", un autre avec que des "2", etc.)
41: 
42:     exemple:
43:     -------
44:     ma_pioche = pioche()
45:     print(ma_pioche)            # affiche la pioche (contenu des 9 sacs de pioche)
46:     print(ma_pioche.contenu)    # affiche la pioche sous forme de liste
47:     print(ma_pioche[1])        # affiche le contenu du premier sac de pioche
48: 
49: 
50:     """
51:     NBR_SACS = 9
52: 
53:     def __init__(self):
54:         """
55:         Initialisation d'une liste contenant 9 sacs
56: 
57:         chaque sac contenant chacun 9 symboles identiques
58:         """
59:         self.contenu = [sac(numéro) for numéro in range(1, self.NBR_SACS+1)]
60: 
61:     def __getitem__(self, index):
62:         """
63:         Permet d'obtenir le contenu d'un sac dans la pioche avec:
64:         ma_pioche[index] # où index est compris entre 1 et NBR_SACS.
65:         """
66:         return self.contenu[index-1]
67: 
68:     def __setitem__(self, index, sac):
69:         """
70:         Permet d'écrire un sac dans la pioche.
71: 
72:         exemple:
73:         -------
74:         ma_pioche[9] = sac(9)
75:         """
76:         if index < self.NBR_SACS+1:
77:             self.contenu[index-1] = sac
78:         else:
79:             raise IndexError()
80: 
81:     def __len__(self):
82:         return self.NBR_SACS
83: 
84:     def __repr__(self):
85:         """
86:         Représentation de la pioche.
87: 
88:         Le contenu des 9 sacs est renvoyé.
89:         """
90:         affichage = ""
91:         for sac in self.contenu:
92:             affichage += sac.__repr__() + "\n"
93:         return affichage
94: 
95: 
96: pioche_sudoku = pioche()
97: print(pioche_sudoku)

La grille comporte 81 cases. 

  1: class SDcolors:
  2:     """
  3:     Gestion basique de la couleur dans le Terminal
  4: 
  5:     Remarque: Toujours terminer par RESET pour revenir à la couleur de base.
  6:     """
  7:     RESET = '\033[0m'
  8:     BOLD = '\033[1m'
  9:     ITALIC = '\033[3m'
 10:     UNDERLINE = '\033[4m'
 11:     GREEN = '\033[5m'
 12:     ORANGE = '\033[6m'
 13:     RED = '\033[7m'
 14:     BLACK = '\033[30m'
 15:     BROWN = '\033[31m'
 16:     LIGHTGREEN = '\033[32m'
 17:     GOLD = '\033[33m'
 18:     BLUE = '\033[34m'
 19:     PURPLE = '\033[35m'
 20:     CYAN = '\033[36m'
 21:     SNOW = '\033[37m'
 22:     BLACKBG = '\033[40m'
 23:     REDBG = '\033[41m'
 24:     GREENBG = '\033[42m'
 25:     GOLDBG = '\033[43m'

 28:     CYANBG = '\033[46m'
 29:     SNOWBG = '\033[47m'

 32: class case:
 33:     """
 34:     Classe représentant une case.
 35: 
 36:     Une case non vide a un `contenu`, le symbole qui est affiché quand on tape
 37:     le nom de la case dans l'interpréteur.
 38:     Une case vide à des prétendants (valeurs possibles de la case).
 39:     Une case a une `couleur` à choisir parmi `SDcolors`.
 40: 
 41:     exemple:
 42:     -------
 43:     ma_case = case()
 44:     print(ma_case)
 45:     """
 46:     def __init__(self):
 47:         self.contenu = None
 48:         self.couleur = SDcolors.BLACK
 49:         self.pretendants = [1, 2, 3, 4, 5, 6, 7, 8, 9]
 50: 
 51:     def __repr__(self):
 52:         """
 53:         Affichage d'une case.
 54: 
 55:         Lorsque l'on tape son nom dans l'interpréteur
 56:         son `contenu` est affiché.
 57:         """
 58:         if self.contenu is None:
 59:             return "0"    # "⛶"
 60:         else:
 61:             return "{}".format(self.contenu)

 64: class grille:
 65:     """
 66:     Classe représentant une grille de 9 x 9 cases.
 67: 
 68:     Chacune des 81 cases est accessible via un index allant de 0 à 80.
 69: 
 70:     exemple:
 71:     -------
 72:     ma_grille = grille()
 73:     print(ma_grille)            # affiche la grille 9 x 9 cases
 74:     print(ma_grille.contenu)    # affiche la grille sous forme de liste
 75:     print(ma_grille[0]) # affiche la première case (son index est 0)
 76: 
 77:     options:
 78:     -------
 79:     Affichage en couleur (fonctionne uniquement dans le Terminal pas dans la
 80:     console Python): ma_grille.affichage_en_couleur = True
 81:     """
 82: 
 83:     NBR_CASES = 81
 84: 
 85:     def __init__(self):
 86:         liste_cases = list()
 87:         for i in range(self.NBR_CASES):
 88:             liste_cases.append(case())
 89:         self.contenu = liste_cases
 90:         self.affichage_en_couleur = False
 91:         self.couleur_grille = SDcolors.SNOWBG
 92: 
 93:     def __getitem__(self, index):
 94:         """
 95:         Permet d'obtenir le contenu d'une case de la grille avec:
 96:         ma_grille[index] # où index est compris entre 0 et NBR_CASES-1.
 97:         """
 98:         return self.contenu[index]
 99: 
100:     def __setitem__(self, index, symbole):
101:         """
102:         Permet d'écrire dans le contenu d'une case de la grille.
103: 
104:         exemple:
105:         -------
106:         ma_grille[0] = 5
107:         """
108:         if index < self.NBR_CASES:
109:             self.contenu[index].contenu = symbole
110:         else:
111:             raise IndexError()
112: 
113:     def __len__(self):
114:         return self.NBR_CASES
115: 
116:     def __repr__(self):
117:         """
118:         Affichage d'une grille.
119: 
120:         Lorsque l'on tape son nom dans l'interpréteur
121:         son `contenu` est affiché sous forme d'une grille 9 x 9.
122:         """
123:         affichage = ""
124:         index = 0
125:         if self.affichage_en_couleur :
126:             couleur_on = self.couleur_grille
127:             couleur_off = SDcolors.RESET
128:         else :
129:             couleur_on, couleur_off = "",""
130:         for une_case in self.contenu:
131:             affichage += une_case.__repr__()  # ajout de l'affichage d'une case
132:             if index % 27 == 26 and index < 80:
133:                 affichage += "\n"+couleur_on + " "*19 + couleur_off + "\n"  # à faire toutes les 3 lignes
134:             elif index % 9 == 8:
135:                 affichage += "\n"  # sinon à faire toutes les lignes
136:             elif index % 3 == 2:
137:                 affichage += couleur_on + "  " + couleur_off  # sinon à faire toutes les 3 colonnes
138:             elif index % 9 in [0, 1, 3, 4, 6, 7]:
139:                 affichage += " "
140:             index += 1
141:         return affichage
142: 
143:     def remplir_case(self, index, valeur):
144:         """
145:         Rempli la case d'index compris entre 0 et 80 avec `valeur`.
146:         """
147:         if self.autorisation_colonne(index) and self.autorisation_ligne(index) and self.autorisation_carré(index):
148:             self.__setitem__(index, valeur)
149:             self.réduire_prétendants
150:             self.réduire_sac
151: 
152:     def autorisation_colonne(self, index):
153:         return True
154: 
155: 
156:     def autorisation_ligne(self, index):
157:         return True
158: 
159:     def autorisation_carré(self, index):
160:         return True
161: 
162:     def réduire_prétendants(self, index):
163:         pass
164: 
165:     def réduire_sac(self, index):
166:         pass
167: 
168:     def colonne(self, index):
169:         """
170:         Retourne le numéro de colonne de la case d'index compris entre 0 et 80.
171: 
172:         Les 9 colonnes sont numérotées de 1 à 9.
173:         """
174:         return index % 9 + 1
175: 
176:     def ligne(self, index):
177:         """
178:         Retourne le numéro de ligne de la case d'index compris entre 0 et 80.
179: 
180:         Les 9 lignes sont numérotées de 1 à 9.
181:         """
182:         return index//9 + 1
